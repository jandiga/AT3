BEGIN Database_Schema
  TABLE Users
    userID (PK)
    email (UNIQUE)
    passwordHash
    role  // "Student" or "Teacher"
    permissions[]
    classCode
    verified BOOL
    linkedPlayerID (FK, NULL allowed)  // Optional link to a player

  TABLE Players
    playerID (PK)
    name
    classCode
    createdByTeacherID (FK)  // Reference to teacher who created the player
    linkedUserID (FK, NULL allowed)  // Optional link to a user account
    academicHistory[]
    effortHistory[]
    totalScore
    isEligibleForDraft BOOL
    dateCreated TIMESTAMP

  TABLE Teams
    teamID (PK)
    leagueID (FK)
    teamName
    playerIDs[]
    managerUserID (FK)
    cumulativeScore

  TABLE Leagues
    leagueID (PK)
    name
    classCode
    createdByTeacherID (FK)
    draftPool[]
    teams[]
    schedule
END Database_Schema
